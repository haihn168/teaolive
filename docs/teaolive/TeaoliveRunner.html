void run(){
  if(_reporter == null){
    _reporter = new TeaoliveReporter();
  }
  
  _reporter.onRunnerStart();
  
  for(TeaoliveTestHolder holder in tests){
    if(holder.isSuite()){
      _executeSuite(holder.suite);
    } else {
      _executeSpec(holder.spec);
    }
  }
  
  _reporter.onRunnerResult(this);
}
